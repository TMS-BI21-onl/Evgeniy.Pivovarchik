SELECT YEAR, [1] AS 'January', [2] AS 'February', [12] AS 'December'
FROM  
(
  SELECT OrderQty, YEAR(DueDate) AS YEAR, MONTH(DueDate) AS MONTH
  FROM Production.WorkOrder
) AS SourceTable  
PIVOT  
(  
  SUM(OrderQty)  
  FOR MONTH IN ([1], [2], [12])  
) AS PivotTable;
