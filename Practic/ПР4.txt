SELECT TOP 5 U.name, COUNT(O.product_id)
FROM user U
LEFT JOIN order O ON u.ID = O.user_id
GROUP BY U.name

5.2

SELECT TOP 5 WITH TIES U.name
(SELECT U.name, COUNT(O.product_id) OVER(PARTITION BY U.id) AS sales
FROM user U
LEFT JOIN order O ON u.ID = O.user_id) s
ORDER BY sales DESC
